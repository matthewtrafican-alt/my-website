Navigation & Files

Shell
# Where am I / move around
pwd  |  cd foldername  |  cd ..  |  cd ~  |  D:
explorer .                # Open in File Explorer
code .                    # Open in VS Code
# List files
ls  |  ls -Force  |  ls -Recurse  |  ls *.txt
# Create / copy / move / delete
mkdir foldername  |  New-Item file.txt
Copy-Item -Recurse src dstMove-Item old.txt new.txt
Remove-Item -Recurse -Force myfolder# Read / write files
cat file.txt  |  Get-Content file.txt -Tail 20  |  Get-Content file.txt -Wait
"Hello" | Out-File file.txt  |  "Hello" | Add-Content file.txt
# Search text in files
Select-String -Recurse "TODO" *.js
# Open in browser
start index.html

command > output.txt  |  command > out.txt 2>&1  |  command | Tee-Object out.txt

Environment Variables

Shell
$env:PATH  |  Get-ChildItem Env:
$env:MY_VAR = "hello"  # Session only
[System.Environment]::SetEnvironmentVariable("MY_VAR","value","User")  # Persistent
set MY_VAR=hello  |  setx MY_VAR "hello"  |  echo %MY_VAR%  # CMD
