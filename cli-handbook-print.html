02

Navigation & File System

Shell 
# Print current directory
pwd                         # PowerShell
cd                          # CMD (no args)

# Change directory
cd C:\Users\You\Projects
cd ..                       # Up one level
cd ~                        # Home directory (PS)
D:                           # Switch to D drive
explorer .                  # Open Explorer here

# List files
ls                          # PowerShell alias
dir                         # CMD and PS
ls -Force                   # Show hidden files
ls -Recurse                 # Recursive listing
ls *.txt                    # Filter by extension
ls | Sort-Object LastWriteTime   # Sort by date modified
03

File Operations

Shell 
# Create
New-Item file.txt
New-Item -ItemType Directory myfolder
mkdir myfolder

# Copy
Copy-Item file.txt backup.txt
Copy-Item -Recurse src\ dst\
copy file.txt backup.txt           # CMD

# Move / Rename
Move-Item old.txt new.txt
Rename-Item file.txt renamed.txt

# Delete
Remove-Item file.txt
Remove-Item -Recurse -Force myfolder\
rmdir /s /q myfolder              # CMD

# Read files
Get-Content file.txt               # (alias: cat, type)
Get-Content file.txt -Tail 20      # Last 20 lines
Get-Content file.txt -Wait         # Live tail

# Write files
"Hello" | Out-File file.txt       # Overwrite
"Hello" | Add-Content file.txt    # Append
"Hello" >> file.txt                # Append (redirect)

# Search in files (grep equivalent)
Select-String "pattern" file.txt
Select-String -Recurse "TODO" *.js
findstr "pattern" file.txt          # CMD
04

Text & Output Manipulation

Shell 
# Filter
ls | Where-Object { $_.Length -gt 1MB }
ls | Where-Object Name -like "*.log"
ls | Where-Object LastWriteTime -gt (Get-Date).AddDays(-7)

# Sort & Select
ls | Sort-Object Length -Descending | Select-Object -First 10

# Export output
ls | Export-Csv output.csv -NoTypeInformation
ls | ConvertTo-Json
ls | Format-Table Name, Length, LastWriteTime

# Redirect
command > output.txt           # Stdout to file
command 2> errors.txt          # Stderr to file
command > out.txt 2>&1        # Both to file
command | Tee-Object out.txt  # Print AND save
05

Environment Variables

Shell 
# Read
$env:PATH
$env:USERNAME
Get-ChildItem Env:             # List all

# Set — current session
$env:MY_VAR = "hello"
$env:PATH += ";C:\new\path"

# Set — persistent (user)
[System.Environment]::SetEnvironmentVariable("MY_VAR", "value", "User")

# Set — persistent (system, admin required)
[System.Environment]::SetEnvironmentVariable("MY_VAR", "value", "Machine")

# CMD equivalents
set                           # List all
set MY_VAR=hello              # Set (session)
setx MY_VAR "hello"          # Set persistent (user)
echo %MY_VAR%                 # Read in CMD

Scripting & Automation

Shell 
# Variables
$name = "World"
$items = @("a", "b", "c")      # Array
$map  = @{ key = "value" }      # Hashtable

# Conditionals  (-eq -ne -gt -lt -like -match)
if ($count -gt 10) { "Big" } elseif ($count -eq 10) { "Ten" } else { "Small" }

# Loops
foreach ($item in $items) { Write-Host $item }
1..10 | ForEach-Object { Write-Host $_ }

# Functions
function Greet($name) { return "Hello, $name!" }

# Error handling
try {
    Get-Item "missing.txt" -ErrorAction Stop
} catch { Write-Host "Error: $_" } finally { "Done" }

# Allow scripts to run
Set-ExecutionPolicy -Scope CurrentUser RemoteSigned

Git & Version Control

Git
Setup
git config --global user.name "Name"
git config --global user.email "e@mail"
git config --global core.editor "code --wait"
git init
git clone https://github.com/user/repo
Status & Log
git status
git log --oneline --graph
git diff
git diff --staged
git diff main..feature
# Stage & Commit
git add .  |  git add -p  (interactive)
git commit -m "message"
git commit --amend --no-edit           # Fix last commit

# Branches
git switch -c feature/login             # Create + switch
git branch -d feature/login             # Delete merged
git merge --no-ff feature/login
git rebase -i HEAD~3                    # Interactive rebase

# Remote
git pull --rebase  |  git push -u origin feature
git push --force-with-lease

# Stash & Undo
git stash  |  git stash pop  |  git stash list
git restore file.txt                   # Discard unstaged
git restore --staged file.txt          # Unstage
git reset HEAD~1                       # Undo commit (keep files)
git reset --hard HEAD~1                # Undo commit (discard)
git revert abc1234                     # Safe undo (new commit)

Node.js & npm

Node.js 
# Project setup
npm init -y                  # Auto-accept defaults
npm install                  # Install from package.json

# Running scripts

# Inspect & secure

npx ts-node script.ts

# pnpm (faster alternative)
npm install -g pnpm
pnpm install  |  pnpm add express  |  pnpm run dev

Python & pip

Python
⚠️ Always use a virtual environment. Never install packages globally unless absolutely necessary.

pip list  |  pip show requests

# Testing & quality
pytest  |  pytest -v  |  pytest -k "test_login"
black .   # Format code
mypy script.py  # Type check

# Poetry (modern project management)

poetry run python script.py  |  poetry shell

Docker

Docker 
# Containers
docker run -d -p 8080:80 --name myapp nginx
docker run -it ubuntu bash               # Interactive terminal
docker run --rm -v C:\data:/app/data nginx # Mount + auto-remove

docker ps  |  docker ps -a
docker stop myapp  |  docker rm -f myapp
docker logs -f myapp
docker exec -it myapp bash               # Shell into running container
docker stats                              # Live resource usage

# Images
docker pull node:20-alpine
docker build -t myapp:latest .
docker image prune -a                    # Remove unused images

# Docker Compose
docker-compose up -d
docker-compose up --build
docker-compose down -v                  # Stop + remove volumes
docker-compose logs -f
docker-compose exec app bash

# Cleanup
docker system prune -a --volumes        # ⚠ Nuclear cleanup
Dockerfile Quick Reference
FROM    node:20-alpine
WORKDIR /app
COPY    package*.json ./
RUN     npm ci
COPY    . .
EXPOSE  3000
CMD     ["node", "server.js"]
