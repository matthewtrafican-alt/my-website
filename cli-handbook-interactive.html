mkdir myfolder              # Works in PS and CMD
ni file.txt                 # Alias for New-Item
Copy-Item / cpCopy files and folders ▾

copy file.txt backup.txt           # CMD
Copy-Item -Recurse src\ dst\       # Folder recursive
cp -r src/ dst/                    # Unix alias (PS)

Remove-Item file.txt
Remove-Item -Recurse -Force myfolder\   # Folder + contents
del file.txt                              # CMD
rmdir /s /q myfolder                     # CMD recursive
Get-Content / catRead file contents ▾ 
Get-Content file.txt
cat file.txt                       # PS alias
type file.txt                      # CMD
Get-Content file.txt -Tail 20      # Last 20 lines
Get-Content file.txt -Wait         # Live tail (like tail -f)
"Hello" | Out-File file.txt       # Write to file
"Hello" | Add-Content file.txt    # Append to file

Text & Output Manipulation

Piping & FilteringChain and transform output ▾ 
# Filter by condition
ls | Where-Object { $_.Length -gt 1MB }
ls | Where-Object Name -like "*.log"

# Sort and select
ls | Sort-Object Length -Descending
ls | Select-Object -First 10

# Export
ls | Export-Csv output.csv -NoTypeInformation
ls | ConvertTo-Json

# Redirect output
command > output.txt           # Stdout to file
command 2> errors.txt          # Stderr to file
command > out.txt 2>&1        # Both to same file
command | Tee-Object out.txt  # Print AND save

Environment Variables

$env: / setxRead, set, persist env vars ▾ 
# Read variables
$env:PATH
$env:USERNAME
Get-ChildItem Env:             # List ALL

# Set (current session only)
$env:MY_VAR = "hello"
$env:PATH += ";C:\new\path"  # Append to PATH

# Set permanently (user-level)
[System.Environment]::SetEnvironmentVariable("MY_VAR","value","User")

# Set permanently (system, requires admin)
[System.Environment]::SetEnvironmentVariable("MY_VAR","value","Machine")

# CMD equivalents
set                           # List all
set MY_VAR=hello              # Set (session)
setx MY_VAR "hello"          # Set permanently (user)
echo %MY_VAR%                 # Read in CMD
